drop table if exists Show;
drop table if exists Season;
drop table if exists Episode;
drop table if exists Actor;
drop table if exists Character;
drop table if exists Message;
drop table if exists Comment;
drop table if exists Watched;
drop table if exists User;
drop table if exists Country;

create table Show(
    idShow integer not null,
    name varchar(32) not null,
    usualAirtime date,
    network varchar(32),
    rating integer not null,
    genre varchar(32)
);

create table Season(
    number integer,
    show varchar(32) references Show(name)
);

create table Episode(
    number integer,
    releaseDate date,
    season integer references Show(number),
);

create table Actor(
    name varchar(32)
);

create table Character(
    name varchar(32),
    actor varchar(32) references Actor(name),
    show varchar(32) references Show(name)
);

create table Message(
    location varchar(32),
    edition integer,
    country varchar(32) not null,
    race_date date,
    gates integer,
    eliminations integer default 1,
    constraint pk_Race primary key(location,edition)
);

create table Comment(
    location varchar(32),
    edition integer,
    country varchar(32) not null,
    race_date date,
    gates integer,
    eliminations integer default 1,
    constraint pk_Race primary key(location,edition)
);

create table Watched(
    location varchar(32),
    edition integer,
    country varchar(32) not null,
    race_date date,
    gates integer,
    eliminations integer default 1,
    constraint pk_Race primary key(location,edition)
);

create table User(
    location varchar(32),
    edition integer,
    country varchar(32) not null,
    race_date date,
    gates integer,
    eliminations integer default 1,
    constraint pk_Race primary key(location,edition)
);

create table Country(
    location varchar(32),
    edition integer,
    country varchar(32) not null,
    race_date date,
    gates integer,
    eliminations integer default 1,
    constraint pk_Race primary key(location,edition)
);
